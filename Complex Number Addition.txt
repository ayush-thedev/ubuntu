#include <iostream>
using namespace std;

class Complex {
    int real, imag;

public:
    // Default constructor
    Complex() {
        real = 0;
        imag = 0;
    }

    // Parameterized constructor
    Complex(int r, int i) {
        real = r;
        imag = i;
    }

    // Friend function for input
    friend istream& operator>>(istream& in, Complex& c);

    // Friend function for addition
    friend Complex operator+(Complex c1, Complex c2);

    // Overload prefix increment ++
    Complex operator++() {
        ++real;
        ++imag;
        return *this;
    }

    // Function to display
    void display() {
        cout << real << " + " << imag << "i" << endl;
    }
};

// Overload >> operator for input
istream& operator>>(istream& in, Complex& c) {
    in >> c.real >> c.imag;
    return in;
}

// Overload + operator using friend function
Complex operator+(Complex c1, Complex c2) {
    Complex temp;
    temp.real = c1.real + c2.real;
    temp.imag = c1.imag + c2.imag;
    return temp;
}

// Main Function
int main() {
    Complex c1, c2, sum;

    cout << "Enter first complex number: ";
    cin >> c1;
    cout << "Enter second complex number: ";
    cin >> c2;

    sum = c1 + c2;

    cout << "Sum = ";
    sum.display();

    ++c1;  // increment first complex number

    cout << "After increment: ";
    c1.display();

    return 0;
}
